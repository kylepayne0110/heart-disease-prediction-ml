Scenario¶

You are a data analyst researching risk factors for heart disease at a university hospital. You have access to a large set of historical data that you can use to analyze patterns between different health indicators (e.g. fasting blood sugar, maximum heart rate, etc.) and the presence of heart disease. You have been asked to create different logistic regression models that predict whether or not a person is at risk for heart disease. A model like this could eventually be used to evaluate medical records and look for risks that might not be obvious to human doctors. You have also been asked to create a classification random forest model to predict the risk of heart disease and a regression random forest model to predict the maximum heart rate achieved.

There are several variables in this data set, but you will be working with the following important variables:

Variable

The person's sex (1 = male, 0 = female)

The person's resting blood pressure
chol
The person's cholesterol measurement in mg/dl

The person's fasting blood sugar is greater than 120 mg/dl (1 = true, 0 = false)

slope
The slope of the peak exercise ST segment (1=upsloping, 2=flat, 3=downsloping)
ca
The number of major vessels (0-3)
target
Heart disease (0=no, 1=yes)

#fit the logistic regression model
model1 <- glm(target ~ age + trestbps + exang + thalach,
              data = heart_data,
              family = "binomial")

#summarize the model
summary(model1)

#predicted probabilities
predicted_prob1 <- predict(model1, type = "response")

#convert predicted probabilities to binary classification using 0.5 threshold
predicted_class1 <- ifelse(predicted_prob1 > 0.5, 1, 0)

#convert to factors with matching levels
predicted_class1 <- factor(predicted_class1, levels = c(0, 1))
actual_class1    <- heart_data$target

#confusion matrix
conf_matrix1 <- table(Predicted = predicted_class1, Actual = actual_class1)
print(conf_matrix1)

#calculate metrics
TP <- conf_matrix1["1", "1"]
TN <- conf_matrix1["0", "0"]
FP <- conf_matrix1["1", "0"]
FN <- conf_matrix1["0", "1"]

accuracy1  <- (TP + TN) / sum(conf_matrix1)
precision1 <- TP / (TP + FP)
recall1    <- TP / (TP + FN)

#print metrics (rounded to four decimals)

#Split the data: 85% training, 15% testing
set.seed(6522048)

#Ensure outcome is a factor
heart_data$target <- factor(heart_data$target)

n_total   <- nrow(heart_data)
samp_size <- floor(0.85 * n_total)
train_ind <- sample(seq_len(n_total), size = samp_size)

train.data <- heart_data[train_ind, ]
test.data  <- heart_data[-train_ind, ]

print("Number of rows in the original data set"); print(n_total)
print("Number of rows in the training set");       print(nrow(train.data))
print("Number of rows in the testing set");        print(nrow(test.data))

#Graph training and testing error against number of trees (max 150)
library(randomForest)
set.seed(6522048)

trees_seq <- 1:150
train_err <- c()
test_err  <- c()

for (i in trees_seq) {
  rf_i <- randomForest(
    target ~ age + sex + cp + trestbps + chol + restecg + exang + ca,
    data = train.data,
    ntree = i
  )
  
  # Training error
  pred_tr  <- predict(rf_i, train.data, type = "class")
  cm_tr    <- table(Prediction = pred_tr, Actual = train.data$target)
  err_tr   <- 1 - (sum(diag(cm_tr)) / sum(cm_tr))
  train_err <- c(train_err, err_tr)
  
  # Testing error
  pred_te  <- predict(rf_i, test.data, type = "class")
  cm_te    <- table(Prediction = pred_te, Actual = test.data$target)
  err_te   <- 1 - (sum(diag(cm_te)) / sum(cm_te))
  test_err <- c(test_err, err_te)
}

#Plot the error curves
plot(trees_seq, train_err, type = "l", ylim = c(0, 1),
     xlab = "Number of Trees", ylab = "Classification Error",
     col = "red", lwd = 2)
lines(trees_seq, test_err, type = "l", col = "blue", lwd = 2)

       col = c("red", "blue"), lty = 1, lwd = 2)
